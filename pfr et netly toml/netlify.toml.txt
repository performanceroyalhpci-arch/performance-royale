[build]
  publish = "."

# Redirection automatique HTTP -> HTTPS (comportement par défaut de
# Netlify sur les domaines *.netlify.app et les domaines custom avec
# certificat géré ; cette règle le rend explicite).
[[redirects]]
  from = "http://*"
  to = "https://:splat"
  status = 301
  force = true

# Headers de sécurité appliqués à toutes les pages.
# C'est ICI, pas dans une balise <meta>, que X-Frame-Options et
# frame-ancestors ont un effet réel dans le navigateur.
[[headers]]
  for = "/*"
  [headers.values]
    X-Frame-Options = "DENY"
    X-Content-Type-Options = "nosniff"
    Referrer-Policy = "strict-origin-when-cross-origin"
    Permissions-Policy = "camera=(), microphone=(), geolocation=()"
    Content-Security-Policy = "default-src 'self'; script-src 'self' 'unsafe-inline'; style-src 'self' 'unsafe-inline' https://fonts.googleapis.com; font-src 'self' https://fonts.gstatic.com; img-src 'self' data: https:; connect-src 'self' https://script.google.com https://script.googleusercontent.com; base-uri 'self'; form-action 'self'; object-src 'none'; frame-ancestors 'none'"

# Cache long pour les assets statiques si tu en ajoutes plus tard
# (images, CSS séparé, etc.) — sans effet sur le HTML lui-même.
[[headers]]
  for = "/*.css"
  [headers.values]
    Cache-Control = "public, max-age=31536000, immutable"

[[headers]]
  for = "/*.js"
  [headers.values]
    Cache-Control = "public, max-age=31536000, immutable"
